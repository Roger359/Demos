      function countSelected(selectObject) {
        let numberSelected = 0;

Bind() Method Example



      let runner = {
        name: "Runner",
        run: function (speed) {
          console.log(this.name + " " + "runs at" + " " + speed + "mph.");
        },
      };
      
      let flyer = {
        name: "Flyer",
        fly: function (speed) {
          console.log(`${this.name} flies at ${speed} mph.`);
        },
      };
      
      let run = runner.run.bind(flyer, 20);
      
      run()

Console output



      Flyer runs at 20mph.
